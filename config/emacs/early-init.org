* startup

#+begin_src emacs-lisp
(defconst dotfiles/fixed-pitch-font "JetBrainsMono Nerd Font")

(defun dotfiles/early-dark-mode-p ()
  "Return non-nil when the current session should start in dark mode."
  (cond
   ((eq system-type 'darwin)
    (string-match-p
     "Dark"
     (shell-command-to-string "/usr/bin/defaults read -g AppleInterfaceStyle 2>/dev/null")))
   (t t)))

(let ((dark-mode (dotfiles/early-dark-mode-p)))
  (setq default-frame-alist
        (append
         `((background-color . ,(if dark-mode "#0d0e1c" "#ffffff"))
           (cursor-color . ,(if dark-mode "#ffffff" "#000000"))
           (font . ,(format "%s-%d" dotfiles/fixed-pitch-font 11))
           (foreground-color . ,(if dark-mode "#ffffff" "#000000")))
         default-frame-alist)
        initial-frame-alist default-frame-alist))

(setq package-enable-at-startup nil
      frame-inhibit-implied-resize t
      frame-resize-pixelwise t
      inhibit-startup-message t
      inhibit-startup-screen t
      initial-scratch-message nil
      native-comp-jit-compilation nil
      native-comp-async-report-warnings-errors 'silent
      ring-bell-function #'ignore
      use-dialog-box nil
      visible-cursor nil)

;; Package-native .eln files from the Nix package set have been unstable on macOS.
(let ((filtered-paths nil))
  (dolist (path native-comp-eln-load-path)
    (unless (string-match-p "/emacs-packages-deps/share/emacs/native-lisp/?$" path)
      (push path filtered-paths)))
  (setq native-comp-eln-load-path (nreverse filtered-paths)))

(blink-cursor-mode -1)
(menu-bar-mode -1)
(scroll-bar-mode -1)
(tool-bar-mode -1)
#+end_src
