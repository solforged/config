#+begin_src emacs-lisp
(setq evil-want-C-i-jump nil
      evil-want-C-u-scroll t
      evil-want-integration t
      evil-want-keybinding nil)

(require 'undo-fu)
(require 'evil)
(evil-mode 1)

(when (fboundp 'evil-set-undo-system)
  (evil-set-undo-system 'undo-fu))

(require 'evil-collection)
(evil-collection-init)

(require 'general)
(general-evil-setup t)

(general-create-definer dotfiles/leader
  :states '(normal visual motion emacs)
  :keymaps 'override
  :prefix "SPC"
  :global-prefix "C-SPC")

(dotfiles/leader
  "." '(find-file :wk "find file")
  "TAB" '(tab-bar-switch-to-recent-tab :wk "last tab")
  "b" '(:ignore t :wk "buffers")
  "bb" '(consult-buffer :wk "switch")
  "bd" '(kill-current-buffer :wk "kill")
  "br" '(revert-buffer :wk "revert")
  "c" '(:ignore t :wk "code")
  "ca" '(eglot-code-actions :wk "code action")
  "cf" '(eglot-format :wk "format")
  "cr" '(eglot-rename :wk "rename")
  "e" '(:ignore t :wk "errors")
  "eb" '(flymake-show-buffer-diagnostics :wk "buffer")
  "en" '(flymake-goto-next-error :wk "next")
  "ep" '(flymake-goto-prev-error :wk "previous")
  "f" '(:ignore t :wk "files")
  "ff" '(find-file :wk "find")
  "fr" '(consult-recent-file :wk "recent")
  "fs" '(save-buffer :wk "save")
  "g" '(:ignore t :wk "git")
  "gb" '(magit-blame-addition :wk "blame")
  "gs" '(magit-status :wk "status")
  "h" '(:ignore t :wk "help")
  "hf" '(describe-function :wk "function")
  "hv" '(describe-variable :wk "variable")
  "o" '(:ignore t :wk "org")
  "oa" '(org-agenda :wk "agenda")
  "oc" '(org-capture :wk "capture")
  "of" '((lambda () (interactive) (find-file org-default-notes-file)) :wk "inbox")
  "p" '(:ignore t :wk "project")
  "pb" '(consult-project-buffer :wk "buffers")
  "pf" '(project-find-file :wk "find file")
  "pp" '(project-switch-project :wk "switch")
  "ps" '(project-switch-to-buffer :wk "switch buffer")
  "q" '(:ignore t :wk "quit")
  "qq" '(delete-frame :wk "quit frame")
  "s" '(:ignore t :wk "search")
  "sb" '(consult-bookmark :wk "bookmark")
  "si" '(consult-imenu :wk "imenu")
  "sl" '(consult-line :wk "line")
  "sp" '(consult-ripgrep :wk "ripgrep")
  "t" '(:ignore t :wk "toggles")
  "tt" '(consult-theme :wk "theme")
  "tw" '(visual-line-mode :wk "wrap")
  "w" '(:ignore t :wk "windows")
  "w-" '(split-window-below :wk "split below")
  "w/" '(split-window-right :wk "split right")
  "wd" '(delete-window :wk "delete")
  "wh" '(windmove-left :wk "left")
  "wj" '(windmove-down :wk "down")
  "wk" '(windmove-up :wk "up")
  "wl" '(windmove-right :wk "right")
  "wo" '(delete-other-windows :wk "only"))
