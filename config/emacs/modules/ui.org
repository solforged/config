(defconst dotfiles/fixed-pitch-font "JetBrainsMono Nerd Font")
(defconst dotfiles/variable-pitch-font
  (when (eq system-type 'darwin)
    "SF Pro Text"))

(defun dotfiles/font-installed-p (font-name)
  "Return non-nil when FONT-NAME is available."
  (and font-name
       (find-font (font-spec :name font-name))))

(defun dotfiles/apply-fonts (&optional frame)
  "Apply the preferred dotfiles fonts to FRAME or the selected frame."
  (with-selected-frame (or frame (selected-frame))
    (when (dotfiles/font-installed-p dotfiles/fixed-pitch-font)
      (set-face-attribute 'default nil :font dotfiles/fixed-pitch-font :height 110)
      (set-face-attribute 'fixed-pitch nil :font dotfiles/fixed-pitch-font :height 1.0))
    (if (dotfiles/font-installed-p dotfiles/variable-pitch-font)
        (set-face-attribute 'variable-pitch nil :font dotfiles/variable-pitch-font :height 1.05)
      (set-face-attribute 'variable-pitch nil :inherit 'default :height 1.0))))

(defun dotfiles/system-dark-mode-p (&optional frame)
  "Return non-nil when FRAME or the current session prefers a dark theme."
  (cond
   ((eq system-type 'darwin)
    (string-match-p
     "Dark"
     (shell-command-to-string "/usr/bin/defaults read -g AppleInterfaceStyle 2>/dev/null")))
   ((with-selected-frame (or frame (selected-frame))
      (eq (frame-parameter nil 'background-mode) 'dark))
    t)
   (t nil)))

(defun dotfiles/apply-theme (&optional frame)
  "Load the preferred Modus theme for FRAME or the current system appearance."
  (with-selected-frame (or frame (selected-frame))
    (let ((theme (if (dotfiles/system-dark-mode-p frame)
                   'modus-vivendi-tinted
                   'modus-operandi-tinted)))
      (unless (eq theme (car custom-enabled-themes))
        (mapc #'disable-theme custom-enabled-themes)
        (load-theme theme t)))))
