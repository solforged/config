(defun dotfiles/project-root ()
  "Return the current project root or `default-directory'."
  (if-let ((project (project-current)))
      (project-root project)
    default-directory))

(defun dotfiles/open-terminal-here ()
  "Open Eat in the current project root."
  (interactive)
  (let ((default-directory (dotfiles/project-root)))
    (call-interactively #'vterm)))

(defun dotfiles/vterm-run (name cmd)
  "Open a vterm buffer named NAME in other window running CMD.
When the process exits, close the window and kill the buffer."
  (let ((vterm-shell cmd))
    (vterm-other-window name))
  (let ((buf (current-buffer))
        (hook nil))
    (setq hook (lambda (buffer _event)
                 (when (eq buffer buf)
                   (remove-hook 'vterm-exit-functions hook)
                   (when-let ((win (get-buffer-window buffer)))
                     (delete-window win))
                   (kill-buffer buffer))))
    (add-hook 'vterm-exit-functions hook)))

(defun dotfiles/claude-here ()
  "Open Claude Code in the current project root."
  (interactive)
  (let ((default-directory (dotfiles/project-root)))
    (dotfiles/vterm-run "*claude*" "claude-bun")))

(defun dotfiles/claude-with-context ()
  "Open Claude Code with current file or region as context."
  (interactive)
  (let* ((default-directory (dotfiles/project-root))
         (file (when (buffer-file-name)
                 (file-relative-name (buffer-file-name) default-directory)))
         (region (when (use-region-p)
                   (buffer-substring-no-properties (region-beginning) (region-end))))
         (cmd (cond
               (region (format "claude-bun %s" (shell-quote-argument
                                            (format "Regarding this code:\n```\n%s\n```" region))))
               (file   (format "claude-bun %s" (shell-quote-argument
                                            (format "Look at %s" file))))
               (t      "claude-bun"))))
    (dotfiles/vterm-run "*claude*" cmd)))

(defvar dotfiles/hive-agents nil
  "Alist of active hive agents: ((NAME . (:worktree :buffer :status :started))).")

(defvar dotfiles/hive-worktree-root ".worktrees"
  "Directory (relative to repo root) where agent worktrees are created.")

(defun dotfiles/hive--repo-root ()
  "Return the git repo root for the current project."
  (let ((root (dotfiles/project-root)))
    (string-trim (shell-command-to-string
                  (format "git -C %s rev-parse --show-toplevel"
                          (shell-quote-argument root))))))

(defun dotfiles/hive--worktree-path (name)
  "Return the worktree path for agent NAME."
  (expand-file-name name
    (expand-file-name dotfiles/hive-worktree-root
      (dotfiles/hive--repo-root))))

(defun dotfiles/hive--buffer-name (name)
  "Return the vterm buffer name for agent NAME."
  (format "*hive:%s*" name))

(defun dotfiles/hive--update-status (name status)
  "Set STATUS for agent NAME and refresh the tab bar."
  (when-let ((agent (assoc name dotfiles/hive-agents)))
    (plist-put (cdr agent) :status status))
  (force-mode-line-update t))

(defun dotfiles/hive--wrap-sentinel (name orig-sentinel)
  "Return a sentinel that calls ORIG-SENTINEL then marks agent NAME as done."
  (lambda (proc event)
    (when orig-sentinel
      (funcall orig-sentinel proc event))
    (when (string-match-p "\\(finished\\|exited\\|killed\\)" event)
      (dotfiles/hive--update-status name 'done)
      (message "Hive: agent '%s' finished" name))))

(defun dotfiles/hive-spawn (name &optional prompt)
  "Spawn a Claude agent in a fresh worktree.
NAME is used for the branch, worktree directory, activity tab, and buffer.
Optional PROMPT is passed to claude as the initial message."
  (interactive "sAgent name: ")
  (when (assoc name dotfiles/hive-agents)
    (if (eq (plist-get (cdr (assoc name dotfiles/hive-agents)) :status) 'done)
        (dotfiles/hive-clean name)
      (user-error "Agent '%s' is already running" name)))
  (let* ((repo-root (dotfiles/hive--repo-root))
         (worktree-dir (dotfiles/hive--worktree-path name))
         (branch (concat "hive/" name))
         (buf-name (dotfiles/hive--buffer-name name))
         (cmd (if prompt
                  (format "claude-bun %s" (shell-quote-argument prompt))
                "claude-bun")))
    ;; Create worktree + branch
    (unless (file-directory-p worktree-dir)
      (let ((err-buf (generate-new-buffer " *hive-git-err*")))
        (unwind-protect
            (let ((exit-code (call-process
                              "git" nil (list nil err-buf) nil "-C" repo-root
                              "worktree" "add" "-b" branch worktree-dir)))
              (unless (zerop exit-code)
                (user-error "Failed to create worktree for '%s': %s"
                            name (string-trim (with-current-buffer err-buf
                                                (buffer-string))))))
          (kill-buffer err-buf))))
    ;; Create an activity tab for this agent
    (activities-new name)
    ;; Open dired + vterm in the worktree
    (let ((default-directory worktree-dir))
      (dired worktree-dir)
      (let ((vterm-shell cmd))
        (vterm buf-name)))
    ;; Track it
    (push (cons name (list :worktree worktree-dir
                           :buffer buf-name
                           :status 'running
                           :started (current-time)))
          dotfiles/hive-agents)
    ;; Watch for exit — chain after vterm's own sentinel so vterm-exit-functions fires
    (when-let ((proc (get-buffer-process buf-name)))
      (set-process-sentinel
       proc (dotfiles/hive--wrap-sentinel name (process-sentinel proc))))
    (message "Hive: spawned agent '%s' on branch %s" name branch)))

(defun dotfiles/hive-spawn-with-prompt (name prompt)
  "Spawn a Claude agent with a task description."
  (interactive "sAgent name: \nsTask prompt: ")
  (dotfiles/hive-spawn name prompt))

(defun dotfiles/hive--completing-read (prompt &optional predicate)
  "Completing-read over hive agent names, optionally filtered by PREDICATE."
  (let ((candidates (if predicate
                        (cl-remove-if-not
                         (lambda (entry) (funcall predicate (cdr entry)))
                         dotfiles/hive-agents)
                      dotfiles/hive-agents)))
    (unless candidates (user-error "No agents"))
    (completing-read prompt (mapcar #'car candidates) nil t)))

(defun dotfiles/hive-switch (name)
  "Switch to agent NAME's activity tab."
  (interactive (list (dotfiles/hive--completing-read "Switch to agent: ")))
  (activities-resume name))

(defun dotfiles/hive-focus (name)
  "Jump to agent NAME's vterm buffer."
  (interactive (list (dotfiles/hive--completing-read "Focus agent: ")))
  (let ((buf-name (plist-get (cdr (assoc name dotfiles/hive-agents)) :buffer)))
    (if (get-buffer buf-name)
        (switch-to-buffer buf-name)
      (user-error "Buffer %s no longer exists" buf-name))))

(defun dotfiles/hive-diff (name)
  "Show the diff for agent NAME's worktree."
  (interactive (list (dotfiles/hive--completing-read "Diff agent: ")))
  (let ((default-directory (plist-get (cdr (assoc name dotfiles/hive-agents)) :worktree)))
    (magit-diff-working-tree "HEAD")))

(defun dotfiles/hive-log (name)
  "Show commits the agent added on its branch."
  (interactive (list (dotfiles/hive--completing-read "Log agent: ")))
  (let ((default-directory (plist-get (cdr (assoc name dotfiles/hive-agents)) :worktree)))
    (magit-log-head)))

(defun dotfiles/hive-clean (name)
  "Tear down agent NAME: kill buffer, remove activity, prune worktree."
  (interactive (list (dotfiles/hive--completing-read "Clean agent: ")))
  (let* ((entry (assoc name dotfiles/hive-agents))
         (props (cdr entry))
         (buf-name (plist-get props :buffer))
         (worktree-dir (plist-get props :worktree))
         (branch (concat "hive/" name)))
    ;; Kill the vterm buffer
    (when-let ((buf (get-buffer buf-name)))
      (when (get-buffer-process buf)
        (delete-process (get-buffer-process buf)))
      (kill-buffer buf))
    ;; Kill the activity
    (ignore-errors (activities-kill name))
    ;; Remove worktree and branch
    (let ((repo-root (dotfiles/hive--repo-root)))
      (when (file-directory-p worktree-dir)
        (call-process "git" nil nil nil "-C" repo-root
                      "worktree" "remove" "--force" worktree-dir))
      (when (zerop (call-process "git" nil nil nil "-C" repo-root
                                 "rev-parse" "--verify" branch))
        (call-process "git" nil nil nil "-C" repo-root
                      "branch" "-D" branch)))
    ;; Forget it
    (setq dotfiles/hive-agents (assoc-delete-all name dotfiles/hive-agents))
    (force-mode-line-update t)
    (message "Hive: cleaned up agent '%s'" name)))

(defun dotfiles/hive-clean-done ()
  "Clean up all agents with 'done' status."
  (interactive)
  (let ((done (cl-remove-if-not
               (lambda (entry) (eq (plist-get (cdr entry) :status) 'done))
               dotfiles/hive-agents)))
    (if (null done)
        (message "Hive: no finished agents to clean")
      (dolist (entry done)
        (dotfiles/hive-clean (car entry))))))

(defun dotfiles/hive-status ()
  "Display a summary of all hive agents in the echo area."
  (interactive)
  (if (null dotfiles/hive-agents)
      (message "Hive: no active agents")
    (message "Hive: %s"
             (mapconcat
              (lambda (entry)
                (let* ((name (car entry))
                       (props (cdr entry))
                       (status (plist-get props :status))
                       (elapsed (float-time
                                 (time-subtract (current-time)
                                                (plist-get props :started))))
                       (mins (truncate (/ elapsed 60))))
                  (format "%s[%s %dm]" name status mins)))
              dotfiles/hive-agents " | "))))

;; Tab-bar segment showing hive agent count
(defun dotfiles/hive-tab-bar-segment ()
  "Return a string for the tab bar showing hive agent status."
  (when dotfiles/hive-agents
    (let ((running (cl-count-if
                    (lambda (e) (eq (plist-get (cdr e) :status) 'running))
                    dotfiles/hive-agents))
          (done (cl-count-if
                 (lambda (e) (eq (plist-get (cdr e) :status) 'done))
                 dotfiles/hive-agents)))
      (concat
       (when (> running 0) (propertize (format " %d" running)
                                        'face '(:foreground "#51afef")))
       (when (> done 0) (propertize (format " %d" done)
                                     'face '(:foreground "#98be65")))))))

(with-eval-after-load 'tab-bar
  (unless (memq 'dotfiles/hive-tab-bar-segment tab-bar-format)
    (setq tab-bar-format
          (append tab-bar-format '(dotfiles/hive-tab-bar-segment)))))
