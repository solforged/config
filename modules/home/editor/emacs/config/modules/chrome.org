(defun dotfiles/quiet-nonediting-buffer ()
  "Tone down editing affordances in presentation-oriented buffers."
  (hl-line-mode -1)
  (setq-local cursor-type nil)
  (setq-local header-line-format nil))

(defun dotfiles/current-project-name ()
  "Return the current project name or nil."
  (when-let ((project (project-current nil)))
    (file-name-nondirectory
     (directory-file-name
      (project-root project)))))

(defun dotfiles/mode-line-state ()
  "Return a short buffer state indicator."
  (cond
   (buffer-read-only "RO")
   ((buffer-modified-p) "**")
   (t "  ")))

(defun dotfiles/mode-line-branch ()
  "Return the current VC branch name or nil."
  (when vc-mode
    (string-trim
     (replace-regexp-in-string
      "\\`[: -]+"
      ""
      (format-mode-line '(vc-mode vc-mode))))))

(defun dotfiles/join-segments (&rest segments)
  "Join non-empty SEGMENTS with restrained spacing."
  (string-join
   (delq nil
         (mapcar
          (lambda (segment)
            (when (and segment (not (string= segment "")))
              segment))
          segments))
   "   "))

(defun dotfiles/align-mode-line (left right)
  "Return a mode line with LEFT on the left and RIGHT right-aligned."
  (let ((reserved-width (max 1 (string-width right))))
    (concat
     " "
     left
     (propertize
      " "
      'display
      `((space :align-to (- right ,(1+ reserved-width)))))
     right
     " ")))

(defun dotfiles/render-mode-line ()
  "Render the minimalist mode line."
  (let ((left
         (dotfiles/join-segments
          (propertize (dotfiles/mode-line-state) 'face 'shadow)
          (propertize "%b" 'face 'mode-line-buffer-id)
          (when-let ((project-name (dotfiles/current-project-name)))
            (propertize project-name 'face 'shadow))))
        (right
         (dotfiles/join-segments
          (format-mode-line mode-name)
          (when-let ((branch (dotfiles/mode-line-branch)))
            (propertize branch 'face 'shadow))
          (format-mode-line "%l:%c"))))
    (dotfiles/align-mode-line
     (format-mode-line left)
     (format-mode-line right))))

(defun dotfiles/render-header-line ()
  "Render the contextual header line."
  (when (or buffer-file-name default-directory)
    (let ((path
           (abbreviate-file-name
            (or (and buffer-file-name (file-name-directory buffer-file-name))
                default-directory))))
      (concat
       " "
       (propertize path 'face 'header-line)
       " "))))

(setq-default mode-line-format '((:eval (dotfiles/render-mode-line)))
              header-line-format '((:eval (dotfiles/render-header-line))))

(add-hook 'help-mode-hook #'dotfiles/quiet-nonediting-buffer)
