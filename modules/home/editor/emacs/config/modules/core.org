* core

#+begin_src emacs-lisp
(setq auto-save-default nil
      completion-cycle-threshold 3
      create-lockfiles nil
      delete-by-moving-to-trash t
      fill-column 100
      indent-tabs-mode nil
      make-backup-files nil
      read-process-output-max (* 1024 1024)
      tab-always-indent 'complete
      tab-width 2
      use-short-answers t)

(setq custom-file (expand-file-name "custom.el" user-emacs-directory))
(load custom-file t)

(defconst dotfiles/org-directory
  (expand-file-name "org" (or (getenv "XDG_DATA_HOME") (expand-file-name "~/.local/share")))
  "Default location for Org data.")

(defconst dotfiles/state-directory
  (expand-file-name "emacs" (or (getenv "XDG_STATE_HOME") (expand-file-name "~/.local/state")))
  "State directory for Emacs runtime files.")

(make-directory dotfiles/org-directory t)
(make-directory dotfiles/state-directory t)

(setq-default cursor-in-non-selected-windows nil)

(global-set-key (kbd "<escape>") #'keyboard-escape-quit)
#+end_src
