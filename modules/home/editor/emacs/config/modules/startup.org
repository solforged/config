#+begin_src emacs-lisp
(require 'button)
(require 'project)
(require 'recentf)
(require 'seq)
(require 'subr-x)

(defconst dotfiles/startup-style
  (dotfiles/getenv-string "DOTFILES_EMACS_STARTUP_STYLE" "minimal-dashboard"))
(defconst dotfiles/startup-buffer-name "*welcome*")

(defvar dotfiles/startup-mode-map
  (let ((map (make-sparse-keymap)))
    (define-key map (kbd "g") #'dotfiles/open-startup)
    (define-key map (kbd "q") #'quit-window)
    (define-key map (kbd "f") #'dotfiles/startup-find-file)
    (define-key map (kbd "p") #'dotfiles/startup-open-project)
    (define-key map (kbd "r") #'consult-recent-file)
    (define-key map (kbd "a") #'org-agenda)
    (define-key map (kbd "c") #'org-capture)
    map)
  "Local keymap for the minimalist startup buffer.")

(define-derived-mode dotfiles/startup-mode special-mode "Dotfiles-Startup"
  "Minimal startup buffer for the dotfiles Emacs UI.")

(defun dotfiles/startup-apply-layout ()
  "Apply restrained layout rules to the startup buffer."
  (setq-local left-margin-width 5)
  (setq-local right-margin-width 0)
  (setq-local truncate-lines t)
  (setq-local line-spacing (+ dotfiles/line-spacing 0.05))
  (when-let ((window (get-buffer-window (current-buffer) t)))
    (set-window-margins window left-margin-width right-margin-width)))

(defun dotfiles/startup-find-file ()
  "Open a file from the startup buffer."
  (interactive)
  (call-interactively #'dotfiles/find-file-dwim))

(defun dotfiles/startup-open-project ()
  "Switch to a project from the startup buffer."
  (interactive)
  (call-interactively #'project-switch-project))

(defun dotfiles/startup-open-inbox ()
  "Open the Org inbox."
  (interactive)
  (find-file org-default-notes-file))

(defun dotfiles/startup-projects ()
  "Return the known project roots for the startup buffer."
  (seq-take
   (if (fboundp 'project-known-project-roots)
       (project-known-project-roots)
     nil)
   5))

(defun dotfiles/startup-recents ()
  "Return recent files that still exist."
  (seq-take
   (seq-filter #'file-exists-p recentf-list)
   6))

(defun dotfiles/insert-startup-heading (title)
  "Insert a section heading TITLE."
  (insert (propertize title 'face 'dotfiles-startup-section))
  (insert "\n\n"))

(defun dotfiles/insert-startup-button (label action &optional face)
  "Insert a startup button with LABEL and ACTION."
  (insert-text-button
   label
   'action action
   'face (or face 'link)
   'follow-link t)
  (insert "\n"))

(defun dotfiles/startup-visit-file-button (button)
  "Visit the file referenced by BUTTON."
  (find-file (button-get button 'dotfiles-path)))

(defun dotfiles/startup-open-project-button (button)
  "Open a project file prompt for the root referenced by BUTTON."
  (let ((default-directory (button-get button 'dotfiles-root)))
    (call-interactively #'project-find-file)))

(defun dotfiles/startup-command-button (button)
  "Invoke the interactive command referenced by BUTTON."
  (call-interactively (button-get button 'dotfiles-command)))

(defun dotfiles/insert-startup-actions ()
  "Insert the high-signal startup actions."
  (dotfiles/insert-startup-heading "Actions")
  (dolist
      (entry
       '(("Find file" . dotfiles/find-file-dwim)
         ("Project" . project-switch-project)
         ("Inbox" . dotfiles/startup-open-inbox)
         ("Agenda" . org-agenda)
         ("Capture" . org-capture)))
    (insert "  ")
    (insert-text-button
     (car entry)
     'action #'dotfiles/startup-command-button
     'dotfiles-command (cdr entry)
     'face 'dotfiles-startup-action
     'follow-link t)
    (insert "\n"))
  (insert "\n"))

(defun dotfiles/insert-startup-recents ()
  "Insert recent-file entries."
  (dotfiles/insert-startup-heading "Recent")
  (if-let ((files (dotfiles/startup-recents)))
      (dolist (file files)
        (insert-text-button
         (abbreviate-file-name file)
         'action #'dotfiles/startup-visit-file-button
         'dotfiles-path file
         'face 'dotfiles-startup-path
         'follow-link t)
        (insert "\n"))
    (insert (propertize "No recent files yet.\n" 'face 'dotfiles-startup-muted)))
  (insert "\n"))

(defun dotfiles/insert-startup-projects ()
  "Insert known projects."
  (dotfiles/insert-startup-heading "Projects")
  (if-let ((projects (dotfiles/startup-projects)))
      (dolist (root projects)
        (insert-text-button
         (abbreviate-file-name root)
         'action #'dotfiles/startup-open-project-button
         'dotfiles-root root
         'face 'dotfiles-startup-path
         'follow-link t)
        (insert "\n"))
    (insert (propertize "No projects tracked yet.\n" 'face 'dotfiles-startup-muted)))
  (insert "\n"))

(defun dotfiles/render-startup-buffer ()
  "Populate the minimalist startup buffer."
  (let ((inhibit-read-only t))
    (erase-buffer)
    (insert "\n\n")
    (insert (propertize "emacs\n" 'face 'dotfiles-startup-title))
    (insert (propertize (format "%s @ %s\n\n\n" user-login-name system-name) 'face 'dotfiles-startup-subtitle))
    (dotfiles/insert-startup-actions)
    (dotfiles/insert-startup-recents)
    (dotfiles/insert-startup-projects)
    (insert (propertize "Keys: f file   p project   c capture   a agenda   g refresh\n" 'face 'dotfiles-startup-muted))
    (goto-char (point-min))
    (dotfiles/startup-apply-layout)))

(defun dotfiles/open-startup ()
  "Return a refreshed minimalist startup buffer."
  (interactive)
  (let ((buffer (get-buffer-create dotfiles/startup-buffer-name)))
    (with-current-buffer buffer
      (dotfiles/startup-mode)
      (dotfiles/quiet-nonediting-buffer)
      (dotfiles/render-startup-buffer))
    buffer))

(setq initial-buffer-choice
      (pcase dotfiles/startup-style
        ("minimal-dashboard" #'dotfiles/open-startup)
        ("blank" (lambda () (get-buffer-create "*scratch*")))
        (_ nil)))
