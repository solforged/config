(defun dotfiles/switch-to-last-buffer ()
  "Switch to the most recently visited buffer."
  (interactive)
  (switch-to-buffer (other-buffer (current-buffer) 1)))

(defun dotfiles/file-minibuffer-tilde ()
  "Reset the minibuffer path to ~/ when tilde is typed after a slash."
  (interactive)
  (if (and (eq (char-before) ?/)
           (minibufferp))
      (let ((end (point))
            (beg (minibuffer-prompt-end)))
        (delete-region beg end)
        (insert "~/"))
    (insert "~")))

(add-hook 'minibuffer-setup-hook
          (lambda ()
            (when (eq this-command #'find-file)
              (local-set-key "~" #'dotfiles/file-minibuffer-tilde))))
