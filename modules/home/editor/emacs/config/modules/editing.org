(defvar dotfiles--file-home-dwim-enabled nil
  "Enable Doom-style home rebasing for `dotfiles/find-file-dwim'.")

(defvar-local dotfiles--file-name-initial-contents nil
  "Initial minibuffer contents captured for filename completion prompts.")

(defun dotfiles/capture-file-name-minibuffer-state ()
  "Remember the initial contents for filename completion prompts."
  (when minibuffer-completing-file-name
    (setq-local dotfiles--file-name-initial-contents
                (minibuffer-contents-no-properties))))

(defun dotfiles/minibuffer-file-home-dwim ()
  "Rebase filename completion to the home directory when the prompt is empty."
  (interactive)
  (if (and dotfiles--file-home-dwim-enabled
           minibuffer-completing-file-name
           (string=
            (minibuffer-contents-no-properties)
            (or dotfiles--file-name-initial-contents "")))
      (progn
        (setq default-directory (expand-file-name "~/"))
        (delete-minibuffer-contents)
        (insert "~/"))
    (self-insert-command 1)))

(define-key minibuffer-local-filename-completion-map
            (kbd "~")
            #'dotfiles/minibuffer-file-home-dwim)

(defun dotfiles/find-file-dwim ()
  "Run `find-file` with a home-directory shortcut on initial `~`."
  (interactive)
  (let ((dotfiles--file-home-dwim-enabled t))
    (minibuffer-with-setup-hook
        #'dotfiles/capture-file-name-minibuffer-state
      (call-interactively #'find-file))))

(defun dotfiles/format-buffer ()
  "Format the current buffer using the configured formatter."
  (interactive)
  (cond
   ((derived-mode-p 'nix-mode)
    (dotfiles/nix-format-buffer))
   ((fboundp 'eglot-format)
    (eglot-format))
   (t
    (user-error "No formatter configured for %s" major-mode))))
